  7. double investmentValue(double investment, double interestRate, int years) {
   if (years == 0)
      return investment;
   else
      return investmentValue(investment * (1 + interestRate), interestRate, years-1);
}


  8. double amountOwed(double loan, double interestRate, double payment, int months) {
   if (months == 0)
      return loan;
   else
      return amountOwed(loan * (1 + interestRate/12) - payment, interestRate, payment, months-1);
}


  9. String duplicate(String s, int i) {
   if (i == 0)
      return "";
   else
      return s + duplicate(s, i-1);
}


  10. String doubleSuccessively(String s, int i) {
   if (i == 0)
      return s;
   else
      return doubleSuccessively(s + s, i-1);
}
